    function classifyAge(age) {
        if (age < 0) {
          return "Invalid Age";
        } else if (age < 13) {
          return "Child";
        } else if (age < 18) {
          return "Teenager";
        } else if (age < 65) {
          return "Adult";
        } else {
          return "Senior";
        }
    }
