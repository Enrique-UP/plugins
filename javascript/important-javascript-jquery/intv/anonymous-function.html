Anonymous functions are functions without names.
Anonymous functions can be used as an argument to other functions or as an immediately invoked function execution.
In JavaScript, an anonymous function is simply a function that does not have a name. 

        let show = function() {
            console.log('Anonymous function');
        };
        show();

        
        // IIFE (Immediately Invoked Function Expression) function
        // An IIFE (Immediately Invoked Function Expression) is a function that runs the moment it is invoked or called in the JavaScript event loop.
        (function() {
            console.log('Anonymous function');
        })();

        
        let person = {
            firstName: 'John',
            lastName: 'Doe'
        };
        (function () {
            console.log(person.firstName + ' ' + person.lastName);
        })(person);


        let add = function (a, b) {
            return a + b;
        };
        // Or
        // let add = (a, b) => a + b;


        // Anonymous function assigned to a variable
        const greet = function(name) {
            return `Hello, ${name}!`;
        };
        console.log(greet("Alice")); // Output: Hello, Alice!

        // Anonymous arrow function
        const add1 = (a, b) => a + b;
        console.log(add1(5, 3)); // Output: 8

        


        const numbers = [1, 2, 3, 4];
        // The function inside map() is anonymous
        const doubled = numbers.map(function(num) {
            return num * 2;
        });
        console.log(doubled); // [ 1, 4, 9, 16 ]
        // Even cleaner with an anonymous arrow function:
        const squared = numbers.map(num => num * num);
        console.log(squared); // Output: [1, 4, 9, 16]



        const button = document.querySelector('button');
        button.addEventListener('click', function() {
            console.log('Button was clicked!');
        });


        (function() {
            const localSecret = "I am hidden!";
            console.log("This runs right away!");
        })();
        // localSecret cannot be accessed out here
