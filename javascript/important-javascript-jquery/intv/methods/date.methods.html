JavaScript Date Methods - Ultimate Cheat Sheet

        console.log("%c====== MASTER DATE METHODS ======", "color: darkorange; font-weight: bold; font-size: 16px;");

        // --- IMPORTANT NOTE ---
        // JS mein Months 0 se shuru hote hain: 0 = January, 1 = February, ..., 11 = December.
        // Days bhi 0 se shuru hote hain: 0 = Sunday, 1 = Monday, ..., 6 = Saturday.

        // Base Date Object create kar rahe hain testing ke liye
        const now = new Date(); // Current date aur time
        console.log("Current Full Date:", now);


        // --- SECTION 1: GET METHODS (Date & Time Nikalne Ke Liye) ---
        console.log("\n--- SECTION 1: GET METHODS ---");

        // 1. getFullYear() - Poora saal (Year) nikal kar deta hai (YYYY)
        console.log(now.getFullYear());              // Output: 2026 (Jaise abhi 2026 chal raha hai)

        // 2. getMonth() - Month nikalta hai (0-11 ke beech)
        console.log(now.getMonth());                 // Output: 6 (Agar July hai toh 6 aayega, kyunki Jan = 0)

        // 3. getDate() - Mahine ki tareeq (Date) nikalta hai (1-31)
        console.log(now.getDate());                  // Output: 17 (Aaj ki date)

        // 4. getDay() - Hafte ka din (Day of the week) nikalta hai (0-6)
        console.log(now.getDay());                   // Output: 5 (Friday ke liye 5, kyunki Sunday = 0)

        // 5. getHours() - Ghante (Hours) nikalta hai (0-23)
        console.log(now.getHours());                 // Output: Current hour (e.g., 18)

        // 6. getMinutes() - Minutes nikalta hai (0-59)
        console.log(now.getMinutes());               // Output: Current minutes (e.g., 45)

        // 7. getSeconds() - Seconds nikalta hai (0-59)
        console.log(now.getSeconds());               // Output: Current seconds

        // 8. getTime() - 1 Jan 1970 se lekar ab tak ka total samay Milliseconds mein deta hai (Timestamp)
        console.log(now.getTime());                  // Output: Ek bada number (e.g., 1784293845000)


        // --- SECTION 2: SET METHODS (Date & Time Badalne/Set Karne Ke Liye) ---
        console.log("\n--- SECTION 2: SET METHODS ---");
        const customDate = new Date(); // Naya object taaki purana change na ho

        // 9. setFullYear() - Date ka saal badal deta hai
        customDate.setFullYear(2030);
        console.log(customDate.getFullYear());       // Output: 2030

        // 10. setMonth() - Month badal deta hai (0-11)
        customDate.setMonth(0);                      // January set kar diya
        console.log(customDate.getMonth());          // Output: 0

        // 11. setDate() - Date (Tareeq) badal deta hai (1-31)
        customDate.setDate(25);
        console.log(customDate.getDate());           // Output: 25
        
        // *Pro Tip: Agar setDate(0) karoge toh yeh pichle mahine ki aakhri tareeq par chala jayega.

        // 12. setHours() - Ghante badal deta hai
        customDate.setHours(10);
        console.log(customDate.getHours());          // Output: 10


        // --- SECTION 3: FORMATTING METHODS (Sundar Format Mein Dikhane Ke Liye) ---
        console.log("\n--- SECTION 3: FORMATTING METHODS ---");

        const testFormat = new Date(2026, 6, 17, 18, 30); // 17 July 2026, 06:30 PM

        // 13. toDateString() - Sirf Date part ko human-readable string mein deta hai (Time hata deta hai)
        console.log(testFormat.toDateString());       // Output: "Fri Jul 17 2026"

        // 14. toTimeString() - Sirf Time part ko string mein deta hai (Date hata deta hai)
        console.log(testFormat.toTimeString());       // Output: "18:30:00 GMT+0530 (India Standard Time)"

        // 15. toLocaleString() - User ke location ke hisab se Date aur Time dono format karta hai
        console.log(testFormat.toLocaleString('en-IN')); // Output: "17/7/2026, 6:30:00 pm" (Indian Format)
        console.log(testFormat.toLocaleString('en-US')); // Output: "7/17/2026, 6:30:00 PM" (US Format)

        // 16. toLocaleDateString() - Sirf Date ko local format mein deta hai
        console.log(testFormat.toLocaleDateString('en-IN')); // Output: "17/7/2026"

        // 17. toISOString() - Standard ISO Format (YYYY-MM-DDTHH:mm:ss.sssZ) mein deta hai (Database mein save karne ke liye best hai)
        console.log(testFormat.toISOString());        // Output: "2026-07-17T13:00:00.000Z" (UTC Time)


        // --- SECTION 4: BONUS static METHOD ---
        console.log("\n--- SECTION 4: BONUS ---");

        // 18. Date.now() - Bina new Date() banaye, seedhe current timestamp (milliseconds) nikalne ke liye
        console.log(Date.now());                      // Output: Current timestamp number
