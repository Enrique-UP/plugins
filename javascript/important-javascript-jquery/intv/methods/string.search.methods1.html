JavaScript String Methods (Search & Verification) Cheat Sheet

        console.log("%c--- JS STRING METHODS (PART 2) ---", "color: purple; font-weight: bold; font-size: 16px;");

        // 1. String indexOf() - Kisi word ya character ka pehla index dhoondta hai
        console.log("\n--- 1. String indexOf() ---");
        console.log("Hello World".indexOf("o"));          // Output: 4
        console.log("Code for Code".indexOf("Code", 1));  // Output: 9 (Index 1 ke baad se dhoondna shuru karega)
        console.log("JavaScript".indexOf("Python"));      // Output: -1 (Agar na mile toh -1 deta hai)

        // 2. String lastIndexOf() - Kisi word ya character ka aakhri index dhoondta hai
        console.log("\n--- 2. String lastIndexOf() ---");
        console.log("Hello World".lastIndexOf("o"));      // Output: 7
        console.log("apple pie apple".lastIndexOf("apple")); // Output: 10
        console.log("Frontend".lastIndexOf("x"));         // Output: -1

        // 3. String includes() - Check karta hai ki word string mein hai ya nahi (true/false)
        console.log("\n--- 3. String includes() ---");
        console.log("Coding is fun".includes("fun"));     // Output: true
        console.log("Coding is fun".includes("Fun"));     // Output: false (Case-sensitive hota hai)
        console.log("JavaScript".includes("Script", 4));   // Output: true (Index 4 ke baad check karega)

        // 4. String startsWith() - Check karta hai kya string usi word se shuru ho rahi hai
        console.log("\n--- 4. String startsWith() ---");
        console.log("Welcome to home".startsWith("Welcome")); // Output: true
        console.log("Welcome to home".startsWith("home"));    // Output: false
        console.log("Welcome to home".startsWith("to", 8));   // Output: true (Index 8 se check karega)

        // 5. String endsWith() - Check karta hai kya string usi word par khatam ho rahi hai
        console.log("\n--- 5. String endsWith() ---");
        console.log("Learning JS".endsWith("JS"));        // Output: true
        console.log("Learning JS".endsWith("Learning"));  // Output: false
        console.log("Learning JS".endsWith("Learning", 8)); // Output: true (Pehle 8 characters mein check karega)

        // 6. String search() - Regular Expression ya string ko search karke uska index deta hai
        console.log("\n--- 6. String search() ---");
        console.log("I love JavaScript".search("Java"));   // Output: 7
        console.log("I love JavaScript".search(/javascript/i)); // Output: 7 (Regex 'i' ignore case flag ke sath)
        console.log("Hello".search("World"));             // Output: -1

        // 7. String match() - Pure match ko array ke roop mein nikalta hai (Regex ke sath)
        console.log("\n--- 7. String match() ---");
        console.log("The Rain in SPAIN stays mainly in the plain".match(/ain/g)); 
        // Output: ["ain", "ain"] (g flag matlab global search)
        
        console.log("Hello 123 World".match(/\d+/g));     
        // Output: ["123"] (\d+ matlab numbers dhoondna)
        
        console.log("Hello".match("World"));              
        // Output: null (Agar match na mile)

        // 8. String matchAll() - Saare matches ka ek Iterator deta hai (Loop chalane ke liye best hai)
        console.log("\n--- 8. String matchAll() ---");
        const iterator = "test1test2".matchAll(/test(\d)/g);
        for (const match of iterator) {
            console.log(match[0]);                        // Output: "test1" aur fir "test2"
        }
        console.log([..."abc".matchAll(/x/g)]);           // Output: [] (Empty array agar koi match na ho)
        console.log([..."hello".matchAll(/l/g)].length);   // Output: 2 (Dono 'l' ke matches ka count)

        // 9. String localeCompare() - Do strings ko alphabetically compare karta hai
        console.log("\n--- 9. String localeCompare() ---");
        console.log("a".localeCompare("b"));              // Output: -1 (Kyunki 'a' pehle aata hai 'b' se)
        console.log("c".localeCompare("b"));              // Output: 1 (Kyunki 'c' baad mein aata hai 'b' se)
        console.log("a".localeCompare("a"));              // Output: 0 (Dono barabar hain)

        // 10. String toString() - Kisi bhi string object ko primitive string mein badalta hai
        console.log("\n--- 10. String toString() ---");
        const strObj = new String("Hello");
        console.log(typeof strObj);                       // Output: "object"
        console.log(typeof strObj.toString());            // Output: "string"
        console.log("Direct String".toString());          // Output: "Direct String"

        // 11. String valueOf() - String ki primitive value return karta hai
        console.log("\n--- 11. String valueOf() ---");
        const valObj = new String("JS");
        console.log(valObj.valueOf());                    // Output: "JS"
        console.log("Hello".valueOf());                   // Output: "Hello"
        console.log(typeof ("test".valueOf()));           // Output: "string"
