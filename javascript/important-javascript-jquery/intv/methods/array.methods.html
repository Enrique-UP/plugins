        console.log(collection); // ['books', 'pens']

		let fruits = ['Apple', 'Banana', 'Cherry', 'Date'];

		}

		let primaryColors = ['Red', 'Blue'];

        // ==========================================
        // 16. filter() - Condition match hone wale elements ka naya array banata hai
        // ==========================================
        const mixNums = [1, 2, 3, 4, 5, 6];
        const evens = mixNums.filter(n => n % 2 === 0); 
        console.log(evens); // [2, 4, 6]

        const adults = ageList.filter(age => age >= 18);

        // ==========================================
        // 17. reduce() - Array ko single value me reduce karta hai (Left to Right)
        // ==========================================
        const expenses = [100, 200, 300];
        const totalExpense = expenses.reduce((acc, curr) => acc + curr, 0); 
        console.log(totalExpense); // 600

        const productOfNums = [1, 2, 3, 4];
        console.log(productOfNums.reduce((acc, curr) => acc * curr, 1)); // 24

        const strings = ['H', 'e', 'l', 'l', 'o'];
        console.log(strings.reduce((acc, curr) => acc + curr, '')); // 'Hello'


        // ==========================================
        // 18. reduceRight() - Right to Left kam karta hai reduce ki tarah
        // ==========================================
        const buildWord = ['o', 'l', 'l', 'e', 'H'];
        console.log(buildWord.reduceRight((acc, curr) => acc + curr, '')); // 'Hello'

        const nested = [[1, 2], [3, 4]];
        console.log(nested.reduceRight((acc, curr) => acc.concat(curr), [])); // [3, 4, 1, 2]

        const subtractRight = [1, 2, 10];
        console.log(subtractRight.reduceRight((acc, curr) => acc - curr)); // 7 (10 - 2 - 1)
