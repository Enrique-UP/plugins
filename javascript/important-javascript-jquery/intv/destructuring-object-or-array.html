// array
        const user = ["Alex", "30", "GM"];
        const [name, age, post] = user;
        console.log(name, age, post); // Alex 30 GM

        // object
        const person = {
            name:"Enrique",
            age:40,
            post:"DM",
            details: function(){
                return `I am ${this.name} ${this.age} years old and in ${this.post}.`;
            }	
        };
        const {name:n, age:a, post:s, details:d, country = "US"} = person;
        console.log(n); // Enrique
        console.log(a); // 40
        console.log(s); // DM
        console.log(country); // US
        console.log(d.call(person)); // I am Enrque 40 years old and in DM.
        console.log(person.details()); // I am Enrque 40 years old and in DM.
