const userSettings = {
            speed: 0,
            graphics: "",
            showTutorial: false,
            difficulty: null // undefined, User hasn't chosen, use default
        };

        // --- Using the OR operator (||) ---
        const speedOr = userSettings.speed || 50; 
        console.log("OR Speed:", speedOr); // 50

        const graphicsOr = userSettings.graphics || "High";
        console.log("OR Graphics:", graphicsOr); // "High"

        // --- Using the Nullish Coalescing operator (??) ---
        // FIXED: Removed '??' from the variable names
        const speedNullish = userSettings.speed ?? 50; 
        console.log("Nullish Speed:", speedNullish); // 0 (Correct! 0 is not null or undefined)

        const graphicsNullish = userSettings.graphics ?? "High"; 
        console.log("Nullish Graphics:", graphicsNullish); // "" (Correct! An empty string is not nullish)

        const difficultyNullish = userSettings.difficulty ?? "Medium"; 
        console.log("Nullish Difficulty:", difficultyNullish); // "Medium" (Correct! The value was null, so it fell back to default)
