// The for...in loop is primarily used for objects to access their property names (keys).
    // for (key in object) {
    //   // code block to be executed
    // }

    const person = {fname:"John", lname:"Doe", age:25}; 
    let txt = "";
    for (let x in person) {
      txt += person[x] + " ";
    }
    console.log(txt); // John Doe 25


    // for (variable in array) {
    //   code

    const numbers = [45, 4, 9, 16, 25];
    let txt1 = "";
    for (let x in numbers) {
      txt1 += numbers[x] + " "; 
    }
    console.log(txt1);
    // 45

    const fruits = ['apple', 'banana', 'orange'];
    for (const index in fruits) {
        console.log(index); // "0", "1", "2"
    }
    for (const fruit of fruits) {
      console.log(fruit); // "apple", "banana", "orange"
    }
    console.log("=============================================================");

    const user = { name: 'Enrique', role: 'CEO' };
    for (const key in user) {
        console.log(key); // "name", "role"
    }
    console.log("---");
    for (const key in user) {
        console.log(user[key]); // "Enrique", "CEO"
    }
    console.log("---");
    for (const value of Object.values(user)){
      console.log(value); // "Enrique", "CEO"
    }
    console.log("---");
    for (const [key, value] of Object.entries(user)){
      console.log(key +" = "+ value);
      // name = Enrique
      // role = CEO
    }
    console.log("=============================================================");
