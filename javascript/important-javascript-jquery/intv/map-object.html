JavaScript Map Objects

Creating a Map from an Array:

    // Create a Map by new Map()
    const fruits = new Map([
      ["apples", 500],
      ["bananas", 300],
      ["oranges", 200]
    ]);
    console.log(fruits); // Map(3) { apples → 500, bananas → 300, oranges → 200 }
    console.log("-----------------------------");
    console.log(fruits.get("apples")); // 500
    console.log("-----------------------------");
    console.log(fruits.size); // 3
  
    console.log("-----------------------------");

    const fruits1 = new Map();
    // Set Map Values
    fruits1.set("apples", 800);
    fruits1.set("bananas", 300);
    fruits1.set("oranges", 200);
    let numb = fruits1.get("apples");
    console.log(numb); // 800
  
    console.log("-----------------------------");

    const letters = new Set(["a","b","c"]);
    letters.add("d");
    letters.add("e");
    letters.add("f");
    // List all Elements
    let text = "";
    letters.forEach (function(value) {
      text += value + " ";
    });
    console.log(text); // a b c d e f
