const name = "Alice";
        const age = 25;
        // Old Way (Concatenation)
        const oldGreeting = "Hello, my name is " + name + " and I am " + age + " years old.";
        console.log(oldGreeting); // Hello, my name is Alice and I am 25 years old.
        // New Way (Template Literals)
        const newGreeting = `Hello, my name is ${name} and I am ${age} years old.`;
        console.log(newGreeting); // Hello, my name is Alice and I am 25 years old.
        
    
        
        
        const price = 100;

        // Ternary operators inside the literal
        const isLoggedIn = true;
