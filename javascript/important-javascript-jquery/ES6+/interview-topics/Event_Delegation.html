        // JavaScript Code
        const parentList = document.getElementById('category-list');

        // सिर्फ पैरेंट (ul) पर एक सिंगल इवेंट लिसनर लगाया
        parentList.addEventListener('click', (event) => {
            
            // event.target हमें वो सटीक एलिमेंट देता है जिसपर असली क्लिक हुआ है
            console.log("असली क्लिक इसपर हुआ:", event.target.id);

            // सुरक्षा के लिए चेक करें कि क्लिक li पर ही हुआ है या नहीं
            if (event.target.tagName === 'LI') {
                alert("आपने चुना: " + event.target.innerText);
            }
        });
