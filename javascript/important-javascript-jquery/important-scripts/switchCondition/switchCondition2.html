const vehicleCode = "M";
    var result;
    switch (vehicleCode) {
        case "C":
            result = "Vehicle Type: Car";
            break;
        case "M":
            result = "Vehicle Type: Motorcycle";
            break;
        case "B":
            result = "Vehicle Type: Bicycle";
            break;
        case "T":
            result = "Vehicle Type: Truck";
            break;
        case "V":
            result = "Vehicle Type: Van";
            break;
        default:
            result = "Invalid vehicle code.";
    }
    document.querySelector("p#demo").innerHTML = result;
